------ QUERY TO CREATE A HN_USERS(HOMIENEST) TABLE -------

CREATE TABLE HN_USERS
(USER_ID VARCHAR2(6) PRIMARY KEY,
 NAME VARCHAR2(20) NOT NULL,
 EMAIL VARCHAR2(50) UNIQUE NOT NULL CHECK(EMAIL LIKE '%@%'),
 PHONE_NO VARCHAR2(10) UNIQUE NOT NULL CHECK(LENGTH(PHONE_NO) = 10),
 ROLE VARCHAR2(15) NOT NULL CHECK(ROLE IN ('BUYER','SELLER','AGENT')),
 CITY VARCHAR2(20) NOT NULL
);

------ QUERY TO CREATE A HN_PROPERTIES TABLE --------

CREATE TABLE HN_PROPERTIES
(PROPERTY_ID VARCHAR2(6) PRIMARY KEY,
 SELLER_ID VARCHAR2(6),
 FOREIGN KEY (SELLER_ID) REFERENCES HN_USERS(USER_ID),
 PROPERTY_TYPE VARCHAR2(10) NOT NULL CHECK(PROPERTY_TYPE IN ('HOUSE','FLAT','LAND')),
 LOCATION VARCHAR2(20) NOT NULL,
 STATUS VARCHAR2(10) NOT NULL CHECK(STATUS IN ('AVAILABLE','SOLD')),
 PRICE NUMBER(10) NOT NULL CHECK(PRICE > 0),
 BEDROOMS NUMBER(3) NOT NULL CHECK(BEDROOMS >= 0),
 BATHROOMS NUMBER(3) NOT NULL CHECK(BATHROOMS >= 0)
);

------ QUERY TO CREATE A HN_BOOKINGS TABLE --------

CREATE TABLE HN_BOOKINGS
(BOOKING_ID VARCHAR2(6) PRIMARY KEY,
 BUYER_ID VARCHAR2(6) NOT NULL,
 FOREIGN KEY (BUYER_ID) REFERENCES HN_USERS(USER_ID),
 PROPERTY_ID VARCHAR2(6) NOT NULL,
 FOREIGN KEY (PROPERTY_ID) REFERENCES HN_PROPERTIES(PROPERTY_ID),
 BOOKING_DATE DATE DEFAULT SYSDATE,
 STATUS VARCHAR2(15) NOT NULL CHECK(STATUS IN ('PENDING','CONFIRMED','CANCELLED'))
);
